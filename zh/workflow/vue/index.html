  <child @hook:mounted="onChildMounted"></child>

  methods: {
    onChildMounted() {}

在 methods 中使用 debounce / throttle ​

vue
<template>
  <div class="container" @click="handleClick"></div>
</template>
<script>
import { debounce } from 'lodash-es'
export default {
  methods: {
    handleClick: debounce(function () {}, 500)
  }
}
</script>
